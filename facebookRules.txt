! === 1. THE AD-STRUCTURE KILLER ===
! Targets the mandatory "About Ads" link hidden in the ad menu.
facebook.com##div[role="feed"] [role="article"]:has(a[href*="/ads/about"]):style(display:none !important;)

! === 2. THE SVG ICON TARGET ===
! FB uses icons instead of text for "Sponsored" now. This nukes those containers.
facebook.com##div[role="feed"] [role="article"]:has(svg):has-text(/S/):has-text(/p/):has-text(/o/):style(display:none !important;)

! === 3. SUGGESTED & GROUPS KILLER ===
! Kills any post with a Join/Follow button or the "Suggested" header.
facebook.com##div[role="feed"] [role="article"]:has(div[aria-label^="Join"]):style(display:none !important;)
facebook.com##div[role="feed"] [role="article"]:has(div[aria-label^="Follow"]):style(display:none !important;)
facebook.com##div[role="feed"] [role="article"]:has-text(/Suggested/):style(display:none !important;)
facebook.com##div[role="feed"] [role="article"]:has-text(/Your group suggestions/):style(display:none !important;)

! === 4. CLEANUP ===
facebook.com##div[aria-label="Stories"]:style(display:none !important;)
facebook.com##div[aria-label="Reels"]:style(display:none !important;)
